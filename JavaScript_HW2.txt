Напишите валидационный скрипт используя функции 

 1. Скрипт должен на вход принимать строку.
 2. После проверки строки выводить в консоль сообщение где будет конкретно написано, что не так в ведённой строке.
 3. Минимум 5 символов в строке
 4. Максимум 64 символа в строке
 5. В строке должны быть буквы
 6. Должна быть хотя бы одна буква в верхнем регистре
 7. Должна быть хотя бы одна цифра
 8. Должна быть хотя бы одна @
 9. Строка не должна быть пустой


function validateString(input) {

if (!input) {
console.log("Строка не должна быть пустой.");
return false;
}

if (input.length < 5) {
console.log("Строка должна содержать минимум 5 символов.");
return false;
}
if (input.length > 64) {
console.log("Строка должна содержать максимум 64 символа.");
return false;
}

if (!/[a-zA-Z]/.test(input)) {
console.log("Строка должна содержать хотя бы одну букву.");
return false;
}

if (!/[A-Z]/.test(input)) {
console.log("Строка должна содержать хотя бы одну букву в верхнем регистре.");
return false;
}

if (!/\d/.test(input)) {
console.log("Строка должна содержать хотя бы одну цифру.");
return false;
}

if (!/@/.test(input)) {
console.log("Строка должна содержать хотя бы один символ '@'.");
return false;
}

console.log("Строка прошла валидацию.");
return true;
}


const testString = "?????";
validateString(testString);
